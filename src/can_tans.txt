#include "config.h"

#define sw1 PA0  // à¸›à¸¸à¹ˆà¸¡à¹€à¸žà¸´à¹ˆà¸¡à¸„à¹ˆà¸² index1

const int txMsgID = 0x069;

const uint32_t txDly = 2000;
uint32_t lastSendTime = 0;
int index1 = 0;
int index2 = 0;

void blink() {
    digitalWrite(PC13, LOW);
    delay(50);
    digitalWrite(PC13, HIGH);
    delay(50);
}

void sendCANMessage() {
    txData[0] = index1;
    txData[1] = index2;

    mySerial.print("ðŸ“¤ Sending CAN ID: 0x");
    mySerial.print(txMsgID, HEX);
    mySerial.print(" | Data: ");
    for (int i = 0; i < 8; i++) {
        mySerial.print("0x");
        if (txData[i] < 0x10) mySerial.print("0");
        mySerial.print(txData[i], HEX);
        mySerial.print(" ");
    }
    mySerial.println();

    bool sent = false;
    for (int i = 0; i < 3; i++) {  // à¸¥à¸­à¸‡à¸ªà¹ˆà¸‡à¸‹à¹‰à¸³à¸ªà¸¹à¸‡à¸ªà¸¸à¸” 3 à¸„à¸£à¸±à¹‰à¸‡
        sent = can.transmit(txMsgID, txData, sizeof(txData));
        if (sent) break;
        mySerial.println("âŒ Failed! Retrying...");
        delay(50);
    }

    if (sent) {
        mySerial.println("âœ… CAN Message Sent!");
        blink();
    } else {
        mySerial.println("ðŸš¨ Critical Error: Resetting CAN Bus...");
        can.begin(STD_ID_LEN, BR250K, PORTA_11_12_XCVR);
    }
}

void setup() {
    mySerial.begin(115200);
    pinMode(PC13, OUTPUT);
    pinMode(sw1, INPUT_PULLUP);
    pinMode(sw2, INPUT_PULLUP);
    pinMode(sw3, INPUT_PULLUP);
    pinMode(sw4, INPUT_PULLUP);
    digitalWrite(PC13, HIGH);
    
    can.begin(STD_ID_LEN, BR250K, PORTA_11_12_XCVR);
    mySerial.println("âœ… CAN Bus Initialized");
    // can.filterMask16Init(0, 0x070, 0x7FF, 0, 0);
}
